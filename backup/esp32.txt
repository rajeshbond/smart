/******************************************************************************
 * FILE        : production_counter_freertos.ino
 * DESCRIPTION : Industrial Production Counter with FreeRTOS
 * BOARD       : ESP32-D0WD-V3
 * VERSION     : 2.7.0 (JSON Queue Fix + Proper Cycle Time)
 ******************************************************************************/

#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <time.h>
#include <Preferences.h>

// =========================================================
// GPIO Configuration
// =========================================================
constexpr uint8_t LH_INPUT_PIN = 18;
constexpr uint8_t RH_INPUT_PIN = 19;

// =========================================================
// Device Configuration
// =========================================================
struct DeviceConfig {
  String tenantId;
  String deviceId;
  String machineId;
  String location;
  String firmwareVersion;
};

DeviceConfig device = {
  "TENANT-001",
  "01@xoom",
  "ASM-001",
  "Assembly Line 01",
  "2.7.0"
};

const String DEVICE_TYPE = "assembly";

// =========================================================
// WiFi Configuration
// =========================================================
const char* WIFI_SSID     = "www.renuscake.com";
const char* WIFI_PASSWORD = "test@1990";

// =========================================================
// MQTT Configuration
// =========================================================
const char* MQTT_SERVER   = "192.168.1.7";
const uint16_t MQTT_PORT  = 1883;

const char* MQTT_USERNAME = "esp32_assembly_01";
const char* MQTT_PASSWORD = "test@1990";

WiFiClient wifiClient;
PubSubClient mqttClient(wifiClient);

// Preferences object for NVS storage
Preferences preferences;

// =========================================================
// MQTT Topics
// =========================================================
String topicProduction;
String topicHeartbeat;
String topicStatus;
String topicCommand;

// =========================================================
// Time Configuration (IST)
// =========================================================
const long GMT_OFFSET_SEC = 19800; // IST = UTC+5:30
const int DAYLIGHT_OFFSET_SEC = 0;

// =========================================================
// Production Variables
// =========================================================
volatile bool lastLHState = HIGH;
volatile bool lastRHState = HIGH;

volatile uint32_t lhCount = 0;
volatile uint32_t rhCount = 0;

volatile unsigned long lastLHCycleTime = 0;
volatile unsigned long lastRHCycleTime = 0;

// =========================================================
// Filter Configuration
// =========================================================
constexpr uint32_t MIN_PRODUCTION_INTERVAL_MS = 20000; // 20 seconds
constexpr uint32_t DEBOUNCE_DELAY_MS = 50;
constexpr uint32_t HEARTBEAT_INTERVAL_MS = 30000;

// =========================================================
// Offline Event Queue (SAFE FIXED BUFFER)
// =========================================================
struct ProductionEvent {
  char payload[256];
};

QueueHandle_t productionQueue;
constexpr uint8_t QUEUE_SIZE = 50;

// =========================================================
// FreeRTOS Handles
// =========================================================
TaskHandle_t inputTaskHandle = NULL;
TaskHandle_t mqttTaskHandle  = NULL;

// =========================================================
// Function Declarations
// =========================================================
void buildTopics();
void connectWiFi();
void connectMQTT();
void initializeTime();
String getTimestamp();
String generateEventId();
void publishHeartbeat();
void queueProduction(const String &station, uint32_t count, float cycleTime);
void processQueuedEvents();
void readProduction(uint8_t pin, const String &station, volatile bool &lastState,
                    volatile uint32_t &counter, volatile unsigned long &lastCycleTime);

// =========================================================
// Build MQTT Topics
// =========================================================
void buildTopics() {
  topicProduction = "factory/" + DEVICE_TYPE + "/assly_production/" + device.deviceId;
  topicHeartbeat  = "factory/" + DEVICE_TYPE + "/heartbeat/" + device.deviceId;
  topicStatus     = "factory/" + DEVICE_TYPE + "/status/" + device.deviceId;
  topicCommand    = "factory/" + DEVICE_TYPE + "/command/" + device.deviceId;
}

// =========================================================
// WiFi Connection
// =========================================================
void connectWiFi() {
  if (WiFi.status() == WL_CONNECTED) return;

  Serial.println("Connecting to WiFi...");

  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected");
}

// =========================================================
// MQTT Callback
// =========================================================
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String json;
  for (unsigned int i = 0; i < length; i++) {
    json += (char)payload[i];
  }

  StaticJsonDocument<256> doc;
  if (deserializeJson(doc, json)) return;

  String command = doc["command"] | "";

  if (command == "RESET_COUNTER") {
    lhCount = 0;
    rhCount = 0;
    lastLHCycleTime = 0;
    lastRHCycleTime = 0;

    preferences.putUInt("lhCount", 0);
    preferences.putUInt("rhCount", 0);

    Serial.println("Counters Reset and Saved to NVS");
  }
}

// =========================================================
// MQTT Connection
// =========================================================
void connectMQTT() {
  mqttClient.setServer(MQTT_SERVER, MQTT_PORT);
  mqttClient.setCallback(mqttCallback);

  if (!mqttClient.connected()) {
    Serial.println("Connecting MQTT...");

    if (mqttClient.connect(device.deviceId.c_str(),
                           MQTT_USERNAME,
                           MQTT_PASSWORD)) {
      Serial.println("MQTT Connected");
      mqttClient.subscribe(topicCommand.c_str());
    } else {
      Serial.print("MQTT Failed: ");
      Serial.println(mqttClient.state());
    }
  }
}

// =========================================================
// Time Initialization
// =========================================================
void initializeTime() {
  configTime(GMT_OFFSET_SEC, DAYLIGHT_OFFSET_SEC,
             "pool.ntp.org", "time.nist.gov");
}

// =========================================================
// Get Timestamp
// =========================================================
String getTimestamp() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) return "";

  char buffer[32];
  strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", &timeinfo);
  return String(buffer);
}

// =========================================================
// Generate Event ID
// =========================================================
String generateEventId() {
  time_t now;
  time(&now);

  uint64_t epochMs = ((uint64_t)now * 1000ULL) + (millis() % 1000);
  return device.deviceId + "-" + String(epochMs);
}

// =========================================================
// Publish Heartbeat
// =========================================================
void publishHeartbeat() {
  if (!mqttClient.connected()) return;

  StaticJsonDocument<256> doc;

  doc["event_id"] = generateEventId();
  doc["tenant_id"] = device.tenantId;
  doc["device_id"] = device.deviceId;
  doc["machine_id"] = device.machineId;
  doc["status"] = "ONLINE";
  doc["rssi"] = WiFi.RSSI();
  doc["timestamp"] = getTimestamp();

  char payload[256];
  serializeJson(doc, payload);

  mqttClient.publish(topicHeartbeat.c_str(), payload, true);
}

// =========================================================
// Queue Production Event (SAFE FIXED BUFFER)
// =========================================================
void queueProduction(const String &station, uint32_t count, float cycleTime) {

  StaticJsonDocument<256> doc;

  doc["event_id"] = generateEventId();
  doc["tenant_id"] = device.tenantId;
  doc["device_id"] = device.deviceId;
  doc["machine_id"] = device.machineId;
  doc["station"] = station;
  doc["count"] = count;
  doc["cycle_time_sec"] = cycleTime;
  doc["timestamp"] = getTimestamp();

  char payload[256];
  serializeJson(doc, payload);

  ProductionEvent event;

  // Copy JSON safely into queue buffer
  strncpy(event.payload, payload, sizeof(event.payload));
  event.payload[sizeof(event.payload) - 1] = '\0';

  if (xQueueSend(productionQueue, &event, 0) != pdPASS) {
    Serial.println("Queue Full! Event Lost");
  } else {
    Serial.println("Event Queued: " + String(event.payload));
  }
}

// =========================================================
// Process Queued Events
// =========================================================
void processQueuedEvents() {
  if (!mqttClient.connected()) return;

  ProductionEvent event;

  while (uxQueueMessagesWaiting(productionQueue) > 0) {
    if (xQueueReceive(productionQueue, &event, 0) == pdPASS) {

      mqttClient.publish(topicProduction.c_str(),
                         event.payload,
                         false);

      Serial.println("Queued Event Published: " + String(event.payload));
    }
  }
}

// =========================================================
// Read Production (Proper Cycle Time Logic)
// =========================================================
void readProduction(uint8_t pin, const String &station,
                    volatile bool &lastState,
                    volatile uint32_t &counter,
                    volatile unsigned long &lastCycleTime) {

  bool currentState = digitalRead(pin);

  if (lastState == HIGH && currentState == LOW) {

    vTaskDelay(DEBOUNCE_DELAY_MS / portTICK_PERIOD_MS);

    if (digitalRead(pin) == LOW) {

      unsigned long currentTime = millis();
      unsigned long cycleTimeMs = currentTime - lastCycleTime;

      // First pulse: initialize timer only
      if (lastCycleTime == 0) {

        lastCycleTime = currentTime;

        Serial.println("========================================");
        Serial.println("FIRST PULSE DETECTED - TIMER INITIALIZED");
        Serial.println("Station: " + station);
        Serial.println("Count: 0");
        Serial.println("Cycle Time: 0.00 sec");
        Serial.println("========================================");

      }
      // Second pulse onward: calculate proper cycle time
      else if (cycleTimeMs >= MIN_PRODUCTION_INTERVAL_MS) {

        float cycleTimeSec = cycleTimeMs / 1000.0;

        counter++;

        // Save counter to NVS immediately
        if (station == "LH") {
          preferences.putUInt("lhCount", counter);
        } else if (station == "RH") {
          preferences.putUInt("rhCount", counter);
        }

        lastCycleTime = currentTime;

        Serial.println("========================================");
        Serial.println("PRODUCTION DETECTED");
        Serial.println("Station: " + station);
        Serial.println("Count: " + String(counter));
        Serial.println("Cycle Time: " + String(cycleTimeSec, 2) + " sec");
        Serial.println("========================================");

        // Queue event for MQTT publishing
        queueProduction(station, counter, cycleTimeSec);

      } else {

        Serial.println("Ignored: Cycle time less than 20 sec");
      }
    }
  }

  lastState = currentState;
}

// =========================================================
// FreeRTOS Input Task
// =========================================================
void inputTask(void *pvParameters) {
  while (1) {
    readProduction(LH_INPUT_PIN, "LH", lastLHState, lhCount, lastLHCycleTime);
    readProduction(RH_INPUT_PIN, "RH", lastRHState, rhCount, lastRHCycleTime);

    vTaskDelay(10 / portTICK_PERIOD_MS);
  }
}

// =========================================================
// FreeRTOS MQTT Task
// =========================================================
void mqttTask(void *pvParameters) {
  unsigned long lastHeartbeat = 0;

  while (1) {

    if (WiFi.status() != WL_CONNECTED) connectWiFi();
    if (!mqttClient.connected()) connectMQTT();

    mqttClient.loop();

    // Publish queued production events
    processQueuedEvents();

    // Heartbeat
    if (millis() - lastHeartbeat >= HEARTBEAT_INTERVAL_MS) {
      lastHeartbeat = millis();
      publishHeartbeat();
    }

    vTaskDelay(100 / portTICK_PERIOD_MS);
  }
}

// =========================================================
// Setup
// =========================================================
void setup() {
  Serial.begin(115200);

  Serial.println("Industrial Production Counter v2.7.0");

  // Initialize Preferences (NVS)
  preferences.begin("prod-counter", false);

  // Restore saved counters
  lhCount = preferences.getUInt("lhCount", 0);
  rhCount = preferences.getUInt("rhCount", 0);

  Serial.println("========================================");
  Serial.println("Restored Counters from NVS");
  Serial.println("LH Count: " + String(lhCount));
  Serial.println("RH Count: " + String(rhCount));
  Serial.println("========================================");

  pinMode(LH_INPUT_PIN, INPUT_PULLUP);
  pinMode(RH_INPUT_PIN, INPUT_PULLUP);

  lastLHState = digitalRead(LH_INPUT_PIN);
  lastRHState = digitalRead(RH_INPUT_PIN);

  buildTopics();
  connectWiFi();
  initializeTime();
  connectMQTT();

  // Create production queue
  productionQueue = xQueueCreate(QUEUE_SIZE, sizeof(ProductionEvent));

  // Create FreeRTOS Tasks
  xTaskCreatePinnedToCore(
    inputTask,
    "InputTask",
    4096,
    NULL,
    2,
    &inputTaskHandle,
    1
  );

  xTaskCreatePinnedToCore(
    mqttTask,
    "MQTTTask",
    8192,
    NULL,
    1,
    &mqttTaskHandle,
    0
  );

  Serial.println("FreeRTOS Tasks Started");
}

// =========================================================
// Main Loop
// =========================================================
void loop() {
  vTaskDelay(1000 / portTICK_PERIOD_MS);
}
